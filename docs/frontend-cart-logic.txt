function initGuestCartId() {
  let guestId = localStorage.getItem("guestCartId");

  if (!guestId) {
    guestId = crypto.randomUUID(); // unique cart id
    localStorage.setItem("guestCartId", guestId);
  }

  return guestId;
}

ðŸŸ¦ 2. Get Auth Headers (auto detects logged in or guest)
function getHeaders() {
  const token = localStorage.getItem("token");          // Logged-in
  const guestCartId = localStorage.getItem("guestCartId"); // Guest

  const headers = {
    "Content-Type": "application/json",
    "guest-cart-id": guestCartId
  };

  if (token) {
    headers["Authorization"] = "Bearer " + token;
  }

  return headers;
}

export async function addToCart(productId, quantity = 1) {
  const response = await fetch("/api/cart/add", {
    method: "POST",
    headers: getHeaders(),
    body: JSON.stringify({ productId, quantity })
  });

  return response.json();
}
